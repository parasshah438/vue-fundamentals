jQuery Complete Guide

Master jQuery from A to Z - Everything you need to know!

What is jQuery?

jQuery is a fast, lightweight, and feature-rich JavaScript library that simplifies HTML document traversing, manipulation, event handling, animation, and Ajax interactions.

Easy DOM Manipulation

Select and manipulate HTML elements with simple syntax

Cross-browser Compatibility

Works consistently across all major browsers

Rich Effects & Animations

Built-in animations and effects

Powerful AJAX Support

Simplified AJAX calls and data handling

Extensive Plugin Ecosystem

Thousands of plugins available

Easy to Learn

Gentle learning curve for beginners

Why jQuery? jQuery simplifies JavaScript programming by providing a consistent API that works across all browsers, reducing development time and complexity.

Where to Use jQuery

  • Web Applications: Interactive websites and web apps
  • Form Handling: Validation, submission, and dynamic forms
  • UI Enhancement: Animations, effects, and user interactions
  • AJAX Applications: Dynamic content loading without page refresh
  • DOM Manipulation: Dynamic content creation and modification
  • Event Handling: User interaction management

Basic Syntax

A. The jQuery Function

// jQuery syntax: $(selector).action()
$(document).ready(function() {
    // Your code here
});

// Shorthand for document ready
$(function() {
    // Your code here
});

// Alternative syntax (avoid conflicts)
jQuery(document).ready(function($) {
    // Your code here
});

B. Basic Structure

// Select elements and perform actions
$('p').hide();              // Hide all paragraphs
$('#myId').show();          // Show element with ID 'myId'
$('.myClass').fadeIn();     // Fade in elements with class 'myClass'

// Chaining methods
$('div').hide().css('color', 'red').fadeIn();

C. The $ Symbol

// $ is an alias for jQuery
$('selector') === jQuery('selector')  // true

// Avoiding conflicts with other libraries
var $j = jQuery.noConflict();
$j(document).ready(function() {
    $j('p').hide();
});
Important: Always wrap your jQuery code in $(document).ready() to ensure the DOM is fully loaded before executing scripts.

Selectors

A. Basic Selectors

// Element selector
$('p')           // All paragraph elements
$('div')         // All div elements

// ID selector
$('#myId')       // Element with ID 'myId'

// Class selector
$('.myClass')    // All elements with class 'myClass'

// Universal selector
$('*')           // All elements

B. Hierarchy Selectors

// Descendant selector
$('div p')       // All p elements inside div elements

// Child selector
$('div > p')     // All p elements that are direct children of div

// Adjacent sibling selector
$('h1 + p')      // p element immediately after h1

// General sibling selector
$('h1 ~ p')      // All p elements that are siblings of h1

C. Attribute Selectors

// Has attribute
$('[title]')           // Elements with title attribute

// Attribute equals
$('[title="hello"]')   // Elements where title="hello"

// Attribute contains
$('[title*="hello"]')  // Elements where title contains "hello"

// Attribute starts with
$('[title^="hello"]')  // Elements where title starts with "hello"

// Attribute ends with
$('[title$="hello"]')  // Elements where title ends with "hello"

D. Pseudo Selectors

// Position-based
$('li:first')       // First li element
$('li:last')        // Last li element  
$('li:even')        // Even-indexed li elements (0, 2, 4...)
$('li:odd')         // Odd-indexed li elements (1, 3, 5...)
$('li:eq(2)')       // Li element at index 2

// State-based
$(':checked')       // All checked form elements
$(':selected')      // All selected option elements
$(':disabled')      // All disabled form elements
$(':enabled')       // All enabled form elements

// Content-based
$(':empty')         // All empty elements
$(':contains("text")')  // Elements containing "text"

// Visibility
$(':visible')       // All visible elements
$(':hidden')        // All hidden elements

E. Form Selectors

// Form element types
$(':input')         // All input, textarea, select, button elements
$(':text')          // All text input elements
$(':password')      // All password input elements
$(':radio')         // All radio button elements
$(':checkbox')      // All checkbox elements
$(':submit')        // All submit button elements
$(':reset')         // All reset button elements
$(':button')        // All button elements
$(':file')          // All file input elements

Event Handling

A. Mouse Events

// Click events
$('button').click(function() {
    alert('Button clicked!');
});

// Double click
$('div').dblclick(function() {
    $(this).hide();
});

// Mouse enter and leave
$('div').mouseenter(function() {
    $(this).css('background-color', 'yellow');
}).mouseleave(function() {
    $(this).css('background-color', '');
});

// Hover (combines mouseenter and mouseleave)
$('div').hover(
    function() { $(this).addClass('highlight'); },    // mouse enter
    function() { $(this).removeClass('highlight'); }  // mouse leave
);

// Mouse down and up
$('button').mousedown(function() {
    $(this).css('background-color', 'red');
}).mouseup(function() {
    $(this).css('background-color', '');
});

B. Keyboard Events

// Key events
$('input').keydown(function(event) {
    console.log('Key pressed: ' + event.which);
});

$('input').keyup(function() {
    console.log('Key released');
});

$('input').keypress(function(event) {
    console.log('Character: ' + String.fromCharCode(event.which));
});

// Specific key detection
$(document).keydown(function(event) {
    if (event.which === 13) { // Enter key
        alert('Enter pressed!');
    }
    if (event.which === 27) { // Escape key
        alert('Escape pressed!');
    }
});

C. Form Events

// Form submission
$('form').submit(function(event) {
    event.preventDefault(); // Prevent default submission
    console.log('Form submitted');
});

// Input focus and blur
$('input').focus(function() {
    $(this).css('border-color', 'blue');
}).blur(function() {
    $(this).css('border-color', '');
});

// Input change
$('input, select').change(function() {
    console.log('Value changed to: ' + $(this).val());
});

// Input event (fires on every keystroke)
$('input').on('input', function() {
    console.log('Current value: ' + $(this).val());
});

D. Event Methods

// Generic event binding
$('button').on('click', function() {
    alert('Clicked!');
});

// Multiple events
$('div').on('mouseenter mouseleave', function() {
    $(this).toggleClass('highlight');
});

// Event delegation (for dynamic elements)
$(document).on('click', '.dynamic-button', function() {
    alert('Dynamic button clicked!');
});

// Remove events
$('button').off('click');

// One-time event
$('button').one('click', function() {
    alert('This will only fire once');
});

// Trigger events programmatically
$('button').trigger('click');

Effects & Animations

A. Basic Show/Hide Effects

// Show and hide
$('#element').show();           // Show element
$('#element').hide();           // Hide element
$('#element').toggle();         // Toggle visibility

// With duration
$('#element').show(1000);       // Show over 1 second
$('#element').hide('slow');     // Hide slowly (600ms)
$('#element').toggle('fast');   // Toggle quickly (200ms)

// With callback
$('#element').hide(500, function() {
    alert('Hide animation complete!');
});

B. Fading Effects

// Fade effects
$('#element').fadeIn();         // Fade in
$('#element').fadeOut();        // Fade out
$('#element').fadeToggle();     // Toggle fade

// Fade to specific opacity
$('#element').fadeTo(1000, 0.5); // Fade to 50% opacity over 1 second

// With duration and callback
$('#element').fadeOut('slow', function() {
    $(this).remove(); // Remove element after fade out
});

C. Sliding Effects

// Slide effects
$('#element').slideDown();      // Slide down
$('#element').slideUp();        // Slide up
$('#element').slideToggle();    // Toggle slide

// With duration
$('#element').slideDown(800);
$('#element').slideUp('fast');

// Slide with easing
$('#element').slideToggle(1000, 'swing', function() {
    console.log('Slide animation complete');
});

D. Custom Animations

// Basic animation
$('#element').animate({
    left: '250px',
    opacity: '0.5',
    height: '150px',
    width: '150px'
}, 1000);

// Complex animation with multiple properties
$('#element').animate({
    left: '+=50px',      // Relative values
    top: '-=25px',
    fontSize: '20px',
    borderRadius: '50%'
}, {
    duration: 2000,
    easing: 'swing',
    complete: function() {
        alert('Animation complete!');
    }
});

// Queue multiple animations
$('#element')
    .animate({left: '100px'}, 500)
    .animate({top: '100px'}, 500)
    .animate({left: '0px'}, 500)
    .animate({top: '0px'}, 500);

// Stop animations
$('#element').stop();           // Stop current animation
$('#element').stop(true);       // Stop and clear queue
$('#element').stop(true, true); // Stop, clear queue, and jump to end

E. Animation Control

// Delay animation
$('#element').delay(1000).fadeOut();

// Check if animating
if ($('#element').is(':animated')) {
    console.log('Element is currently animating');
}

// Custom easing functions (requires jQuery UI or easing plugin)
$('#element').animate({
    left: '100px'
}, 1000, 'easeInOutBounce');

// Animation settings
$.fx.interval = 13;  // Animation frame rate (default: 13ms)
$.fx.off = true;     // Disable all animations
Effect Method Description Duration Options
show() / hide() Show or hide elements 'slow', 'fast', or milliseconds
fadeIn() / fadeOut() Fade elements in or out 'slow', 'fast', or milliseconds
slideDown() / slideUp() Slide elements down or up 'slow', 'fast', or milliseconds
animate() Custom property animations Object with duration, easing, complete
Animation Tips: Use callbacks for chaining animations, stop() method to prevent animation queuing, and consider performance with complex animations.

HTML Manipulation

A. Getting and Setting Content

// Get and set text content
var text = $('p').text();          // Get text content
$('p').text('New text content');   // Set text content

// Get and set HTML content
var html = $('div').html();        // Get HTML content
$('div').html('<strong>Bold text</strong>'); // Set HTML content

// Get and set form values
var value = $('input').val();      // Get input value
$('input').val('New value');       // Set input value

// Get and set attributes
var src = $('img').attr('src');    // Get attribute
$('img').attr('src', 'new.jpg');   // Set attribute
$('img').attr({                    // Set multiple attributes
    'src': 'new.jpg',
    'alt': 'New image',
    'title': 'New title'
});

// Remove attributes
$('img').removeAttr('title');

B. Adding Content

// Append content (inside, at the end)
$('ul').append('<li>New item</li>');
$('<li>New item</li>').appendTo('ul');

// Prepend content (inside, at the beginning)
$('ul').prepend('<li>First item</li>');
$('<li>First item</li>').prependTo('ul');

// Insert content after element
$('p').after('<div>After paragraph</div>');
$('<div>After paragraph</div>').insertAfter('p');

// Insert content before element
$('p').before('<div>Before paragraph</div>');
$('<div>Before paragraph</div>').insertBefore('p');

C. Removing Content

// Remove elements
$('p').remove();                   // Remove all p elements
$('p').remove('.highlight');       // Remove p elements with class 'highlight'

// Empty elements (remove content but keep element)
$('div').empty();

// Detach elements (remove but keep data and events)
var detached = $('p').detach();    // Remove and store
$('body').append(detached);        // Re-add later

D. Replacing Content

// Replace elements
$('p').replaceWith('<div>New content</div>');
$('<div>New content</div>').replaceAll('p');

// Wrap elements
$('p').wrap('<div class="wrapper"></div>');     // Wrap each p individually
$('p').wrapAll('<div class="container"></div>'); // Wrap all p elements together
$('p').wrapInner('<strong></strong>');          // Wrap content inside each p

// Unwrap elements
$('p').unwrap();                   // Remove parent wrapper

E. Cloning Elements

// Clone elements
var cloned = $('div').clone();           // Clone without events
var clonedWithEvents = $('div').clone(true); // Clone with events

// Clone and append
$('div').clone().appendTo('body');

// Clone with data
$('div').clone(true, true).appendTo('body'); // Clone with events and data
Best Practices: Use text() for plain text to prevent XSS attacks, use html() only when you need to insert HTML markup, and always validate user input.

DOM Traversing

A. Tree Traversal

// Parent traversal
$('li').parent();           // Direct parent
$('li').parents();          // All ancestors
$('li').parents('ul');      // Ancestors that match selector
$('li').parentsUntil('div'); // Ancestors until div (exclusive)

// Child traversal
$('ul').children();         // Direct children
$('ul').children('li');     // Direct children that match selector
$('ul').find('a');          // All descendants that match selector

// Sibling traversal
$('li').siblings();         // All siblings
$('li').siblings('.active'); // Siblings that match selector
$('li').next();             // Next sibling
$('li').nextAll();          // All following siblings
$('li').nextUntil('.last'); // Following siblings until .last
$('li').prev();             // Previous sibling
$('li').prevAll();          // All preceding siblings
$('li').prevUntil('.first'); // Preceding siblings until .first

B. Filtering

// Filter elements
$('li').first();            // First element in set
$('li').last();             // Last element in set
$('li').eq(2);              // Element at index 2
$('li').filter('.active');  // Elements that match selector
$('li').filter(function(index) {
    return $(this).text().length > 10;
});

// Exclude elements
$('li').not('.disabled');   // Elements that don't match selector
$('li').not(function(index) {
    return $(this).hasClass('hidden');
});

// Check if elements match
if ($('div').is('.active')) {
    console.log('Element has active class');
}

// Get subset
$('li').slice(1, 3);        // Elements from index 1 to 2

C. Collection Manipulation

// Add elements to collection
$('div').add('p');          // Add all p elements
$('div').add('.highlight'); // Add elements with class

// Get elements by index
$('li').get(0);             // Get DOM element (not jQuery object)
$('li')[0];                 // Same as above
$('li').eq(0);              // Get jQuery object

// Iterate through elements
$('li').each(function(index, element) {
    console.log(index + ': ' + $(element).text());
});

// Map elements to array
var texts = $('li').map(function() {
    return $(this).text();
}).get(); // Convert to regular array

// Get collection length
var count = $('li').length;

D. Content Search

// Find elements containing text
$('p:contains("jQuery")');  // Paragraphs containing "jQuery"

// Find empty elements
$('div:empty');

// Find elements with specific attributes
$('input[type="text"]');
$('a[href*="jquery"]');

// Find form elements
$(':input');                // All form elements
$(':text');                 // Text inputs
$(':checkbox:checked');     // Checked checkboxes

Ancestor Methods

  • parent() - Direct parent
  • parents() - All ancestors
  • parentsUntil() - Ancestors until selector

Descendant Methods

  • children() - Direct children
  • find() - All descendants

Sibling Methods

  • siblings() - All siblings
  • next() / prev() - Adjacent siblings
  • nextAll() / prevAll() - All following/preceding

Filtering Methods

  • filter() - Elements that match
  • not() - Elements that don't match
  • first() / last() - First/last element

AJAX

A. Basic AJAX Methods

// GET request
$.get('api/data.json', function(data) {
    console.log('Data loaded:', data);
}).fail(function() {
    console.log('Error loading data');
});

// POST request
$.post('api/submit.php', {
    name: 'John',
    email: 'john@example.com'
}, function(response) {
    console.log('Data submitted:', response);
});

// Load HTML content
$('#content').load('page.html');
$('#content').load('page.html #section'); // Load specific section

// Get JSON data
$.getJSON('api/users.json', function(users) {
    $.each(users, function(index, user) {
        $('ul').append('<li>' + user.name + '</li>');
    });
});

// Get script
$.getScript('js/plugin.js', function() {
    console.log('Script loaded and executed');
});

B. Advanced AJAX with $.ajax()

// Full AJAX configuration
$.ajax({
    url: 'api/data.php',
    type: 'POST',
    dataType: 'json',
    data: {
        id: 123,
        action: 'update'
    },
    headers: {
        'X-API-Key': 'your-api-key'
    },
    timeout: 5000,
    beforeSend: function(xhr) {
        // Show loading indicator
        $('#loading').show();
    },
    success: function(data, status, xhr) {
        console.log('Success:', data);
        $('#result').html(data.message);
    },
    error: function(xhr, status, error) {
        console.log('Error:', error);
        alert('Request failed: ' + error);
    },
    complete: function(xhr, status) {
        // Hide loading indicator
        $('#loading').hide();
    }
});

// Using promises (jQuery 3.0+)
$.ajax({
    url: 'api/data.json',
    method: 'GET'
}).done(function(data) {
    console.log('Success:', data);
}).fail(function(xhr, status, error) {
    console.log('Error:', error);
}).always(function() {
    console.log('Request completed');
});

C. AJAX Global Events

// Global AJAX event handlers
$(document).ajaxStart(function() {
    $('#loading').show();
});

$(document).ajaxStop(function() {
    $('#loading').hide();
});

$(document).ajaxSuccess(function(event, xhr, settings) {
    console.log('AJAX request succeeded');
});

$(document).ajaxError(function(event, xhr, settings, error) {
    console.log('AJAX request failed: ' + error);
});

// Disable global events for specific request
$.ajax({
    url: 'api/data.json',
    global: false  // This request won't trigger global events
});

D. Form Serialization

// Serialize form data
var formData = $('form').serialize();
// Result: "name=John&email=john%40example.com"

var formArray = $('form').serializeArray();
// Result: [{name: "name", value: "John"}, {name: "email", value: "john@example.com"}]

// Submit form via AJAX
$('form').submit(function(e) {
    e.preventDefault();
    
    $.ajax({
        url: 'submit.php',
        type: 'POST',
        data: $(this).serialize(),
        success: function(response) {
            alert('Form submitted successfully!');
        },
        error: function() {
            alert('Form submission failed!');
        }
    });
});

// File upload with FormData
$('#file-form').submit(function(e) {
    e.preventDefault();
    
    var formData = new FormData(this);
    
    $.ajax({
        url: 'upload.php',
        type: 'POST',
        data: formData,
        processData: false,
        contentType: false,
        success: function(response) {
            console.log('File uploaded:', response);
        }
    });
});

E. AJAX Configuration

// Set global AJAX defaults
$.ajaxSetup({
    timeout: 10000,
    cache: false,
    headers: {
        'X-Requested-With': 'XMLHttpRequest'
    }
});

// CSRF token handling
$.ajaxSetup({
    beforeSend: function(xhr, settings) {
        if (!/^(GET|HEAD|OPTIONS|TRACE)$/i.test(settings.type) && !this.crossDomain) {
            xhr.setRequestHeader("X-CSRFToken", $('[name=csrfmiddlewaretoken]').val());
        }
    }
});
Security Note: Always validate and sanitize data on the server side. Never trust client-side data. Use CSRF protection for state-changing requests.

B. Object Utilities

// Extend objects (shallow copy)
var obj1 = {a: 1, b: 2};
var obj2 = {c: 3, d: 4};
var extended = $.extend(obj1, obj2); // obj1 is modified

// Deep extend (deep copy)
var obj1 = {a: {x: 1, y: 2}};
var obj2 = {a: {z: 3}};
var deepExtended = $.extend(true, {}, obj1, obj2);

// Check if object is empty
$.isEmptyObject({}); // true
$.isEmptyObject({a: 1}); // false

// Check if plain object
$.isPlainObject({}); // true

    var methods = {
        init: function(options) {
            var settings = $.extend({
                speed: 300,
                color: 'red'
            }, options);
            
            return this.each(function() {
                $(this).data('pluginData', settings);
            });
        },
        
        destroy: function() {
            return this.each(function() {
                $(this).removeData('pluginData');
            });
        },
        
        update: function(options) {
            return this.each(function() {
                var settings = $(this).data('pluginData');
                $.extend(settings, options);
            });
        }
    };

        if (methods[method]) {
            return methods[method].apply(this, Array.prototype.slice.call(arguments, 1));
        } else if (typeof method === 'object' || !method) {
            return methods.init.apply(this, arguments);
        } else {
            $.error('Method ' + method + ' does not exist');
        }
    };
})(jQuery);

// Throttle/debounce expensive operations
var throttle = function(func, delay) {
    var timeoutId;
    var lastExecTime = 0;
    return function() {
        var context = this;
        var args = arguments;
        var currentTime = Date.now();
        
        if (currentTime - lastExecTime > delay) {
            func.apply(context, args);
            lastExecTime = currentTime;
        } else {
            clearTimeout(timeoutId);
            timeoutId = setTimeout(function() {
                func.apply(context, args);
                lastExecTime = Date.now();
            }, delay);
        }
    };
};

$(window).on('scroll', throttle(function() {
    // Expensive scroll handler
}, 100));

    }
    
    function loadPage(url) {
        $.ajax({
            url: url,
            method: 'GET',
            dataType: 'html',
            success: handlePageLoad,
            error: handleLoadError
        });
    }
    
    function handlePageLoad(data) {
        $('#main-content').html(data);
    }
    
    function handleLoadError() {
        console.error('Failed to load page');
        // Show user-friendly error message
    }
    
    // Initialize
    initializeNavigation();
});

// Error handling
try {
    // Risky operation
    someFunction();
} catch (error) {
    console.error('Error occurred:', error);
    // Handle error gracefully
}

// Feature detection
if ($.fn.datepicker) {
    $('.date-input').datepicker();
}
