Laravel Livewire v3

Complete Reference Guide - Basic to Advanced

Installation

Laravel Livewire v3 Requirements: Laravel 10+ and PHP 8.1+

Fresh Installation

# Install Livewire v3 via Composer
composer require livewire/livewire

# Publish Livewire config (optional)
php artisan livewire:publish --config

# Publish Livewire assets (optional)
php artisan livewire:publish --assets

Blade Layout Setup

Add Livewire assets to your main layout file:

<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Your App</title>
    @livewireStyles
</head>
<body>
    {{ $slot }}
    
    @livewireScripts
</body>
</html>

Auto-Injection

Livewire v3 can automatically inject assets if you're not using a build tool.

Vite Integration

Perfect integration with Laravel's Vite bundler for optimized assets.

Quick Start

Create Your First Component

# Generate a new Livewire component
php artisan make:livewire Counter

# Or create in a subdirectory
php artisan make:livewire Admin/UsersList

<livewire:counter />

@livewire('counter')

<livewire:counter :initial-count="10" />

Components

Component Generation Commands

# Basic component
php artisan make:livewire ComponentName

# Component with custom view
php artisan make:livewire ComponentName --view=custom-view

# Inline component (no separate view file)
php artisan make:livewire ComponentName --inline

# Component in subdirectory
php artisan make:livewire Admin/Users/UserProfile

Component Structure

Class Properties

Public properties are automatically available in the view and persist between requests.

public string $name = '';
public array $items = [];
public bool $isVisible = true;

Methods

Public methods can be called from the frontend using wire:click and other directives.

public function save()
{
    // Action logic
}

public function delete($id)
{
    // Delete logic
}

Component Lifecycle

Hook Description Usage
mount() Called when component is first loaded Initialize data, validate parameters
hydrate() Called on every request Set up data for each request
updating() Before property update Validate, transform data
updated() After property update React to changes
render() Return the view Render component template

Properties & Data

Property Types

class UserProfile extends Component
{
    // Primitive types
    public string $name = '';
    public int $age = 0;
    public bool $isActive = true;
    public array $tags = [];
    
    // Collections
    public Collection $users;
    
    // Models (must be public)
    public User $user;
    
    // Computed properties
    public function getFullNameProperty()
    {
        return $this->firstName . ' ' . $this->lastName;
    }
}

Data Binding

Two-Way Binding

<!-- Real-time binding -->
<input wire:model.live="name">

<!-- Lazy binding (on blur) -->
<input wire:model.lazy="email">

<!-- Debounced binding -->
<input wire:model.live.debounce.500ms="search">

Form Elements

<!-- Text inputs -->
<input type="text" wire:model="name">

<!-- Select dropdown -->
<select wire:model="category">
    <option value="1">Option 1</option>
</select>

<!-- Checkboxes -->
<input type="checkbox" wire:model="isActive">

Protected vs Public Properties

class ProfileForm extends Component
{
    // Available in view, persistent
    public string $name = '';
    
    // Not available in view, not persistent
    protected array $rules = [
        'name' => 'required|min:3'
    ];
    
    // Exclude from browser (security)
    protected array $hidden = ['password'];
}

Actions & Events

Wire Directives

Directive Description Example
wire:click Handle click events wire:click="save"
wire:submit Handle form submission wire:submit="store"
wire:keydown Handle keyboard events wire:keydown.enter="search"
wire:loading Show loading states wire:loading.class="opacity-50"
wire:target Target specific actions wire:loading wire:target="save"

Method Parameters

<!-- Pass simple parameters -->
<button wire:click="delete({{ $user->id }})">Delete</button>

<!-- Pass multiple parameters -->
<button wire:click="updateStatus({{ $id }}, 'active')">Activate</button>

<!-- Pass dynamic parameters -->
<button wire:click="edit($wire.selectedId)">Edit Selected</button>

Event Dispatching

// In component method
public function save()
{
    // Save logic...
    
    // Dispatch events
    $this->dispatch('user-saved', userId: $this->user->id);
    $this->dispatch('notify', message: 'User saved successfully!');
    
    // Dispatch to specific component
    $this->dispatch('refresh-list')->to(UsersList::class);
    
    // Dispatch browser event (to JavaScript)
    $this->js('alert("Saved!")');
}

Event Listeners

class UsersList extends Component
{
    // Listen to events
    protected $listeners = [
        'user-saved' => 'refreshList',
        'user-deleted' => 'handleUserDeleted',
    ];
    
    // Or using attributes (Laravel 11+)
    #[On('user-saved')]
    public function refreshList($userId = null)
    {
        $this->users = User::all();
    }
}

Forms & Validation

Basic Form Validation

class ContactForm extends Component
{
    public string $name = '';
    public string $email = '';
    public string $message = '';
    
    protected array $rules = [
        'name' => 'required|min:3',
        'email' => 'required|email',
        'message' => 'required|min:10',
    ];
    
    public function submit()
    {
        $this->validate();
        
        // Process form...
        
        session()->flash('message', 'Message sent successfully!');
        $this->reset();
    }
}

Real-time Validation

public function updated($property)
{
    // Validate single property as user types
    $this->validateOnly($property);
}

// Or validate specific properties
public function updatedEmail()
{
    $this->validateOnly('email');
}

Custom Validation Rules

protected function rules()
{
    return [
        'name' => ['required', 'min:3', new CustomRule()],
        'email' => [
            'required', 
            'email', 
            Rule::unique('users')->ignore($this->user->id ?? null)
        ],
    ];
}

protected function messages()
{
    return [
        'name.required' => 'The name field is required.',
        'email.unique' => 'This email is already taken.',
    ];
}

File Uploads

Note: File uploads in Livewire require the WithFileUploads trait.

Basic File Upload

use Livewire\WithFileUploads;

class FileUpload extends Component
{
    use WithFileUploads;
    
    public $photo;
    
    protected array $rules = [
        'photo' => 'image|max:10240', // 10MB Max
    ];
    
    public function save()
    {
        $this->validate();
        
        $path = $this->photo->store('photos', 'public');
        
        // Save to database
        auth()->user()->update(['avatar' => $path]);
        
        session()->flash('message', 'Photo uploaded successfully!');
    }
}

Multiple File Uploads

class MultipleFileUpload extends Component
{
    use WithFileUploads;
    
    public array $photos = [];
    
    protected array $rules = [
        'photos.*' => 'image|max:5120', // 5MB per file
    ];
    
    public function save()
    {
        $this->validate();
        
        foreach ($this->photos as $photo) {
            $path = $photo->store('gallery', 'public');
            
            Gallery::create([
                'user_id' => auth()->id(),
                'path' => $path,
                'filename' => $photo->getClientOriginalName(),
            ]);
        }
        
        $this->photos = [];
        session()->flash('message', 'Photos uploaded successfully!');
    }
}
